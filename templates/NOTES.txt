1. Get the application URL by running these commands:

KRAKEND API GATEWAY:
{{- if eq .Values.krakend.service.type "LoadBalancer" }}
  export KRAKEND_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "ConfiaPe.Loans.CD.fullname" . }}-krakend -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "API Gateway: http://$KRAKEND_IP:{{ .Values.krakend.service.port }}"
{{- else if eq .Values.krakend.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "ConfiaPe.Loans.CD.fullname" . }}-krakend -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[?(@.type=="ExternalIP")].address}')
  echo "API Gateway: http://$NODE_IP:$NODE_PORT"
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "ConfiaPe.Loans.CD.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=krakend" -o jsonpath="{.items[0].metadata.name}")
  echo "API Gateway: http://127.0.0.1:8080"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:{{ .Values.krakend.service.port }}
{{- end }}

  Endpoints:
  - Frontend: /
  - Backend API: /api/*
